229 class dng_warp_params_rectilinear: public dng_warp_params
230  {
231 
232  public:
233 
234  // Radial and tangential polynomial coefficients. These define a warp
235  // from corrected pixel coordinates (xDst, yDst) to uncorrected pixel
236  // coordinates (xSrc, ySrc) for each plane P as follows:
237  //
238  // Let kr0 = fRadParams [P][0]
239  // kr1 = fRadParams [P][1]
240  // kr2 = fRadParams [P][2]
241  // kr3 = fRadParams [P][3]
242  //
243  // kt0 = fTanParams [P][0]
244  // kt1 = fTanParams [P][1]
245  //
246  // Let (xCenter, yCenter) be the optical image center (see fCenter,
247  // below) expressed in pixel coordinates. Let maxDist be the Euclidean
248  // distance (in pixels) from (xCenter, yCenter) to the farthest image
249  // corner.
250  //
251  // First, compute the normalized distance of the corrected pixel
252  // position (xDst, yDst) from the image center:
253  //
254  // dx = (xDst - xCenter) / maxDist
255  // dy = (yDst - yCenter) / maxDist
256  //
257  // r^2 = dx^2 + dy^2
258  //
259  // Compute the radial correction term:
260  //
261  // ratio = kr0 + (kr1 * r^2) + (kr2 * r^4) + (kr3 * r^6)
262  //
263  // dxRad = dx * ratio
264  // dyRad = dy * ratio
265  //
266  // Compute the tangential correction term:
267  //
268  // dxTan = (2 * kt0 * dx * dy) + kt1 * (r^2 + 2 * dx^2)
269  // dyTan = (2 * kt1 * dx * dy) + kt0 * (r^2 + 2 * dy^2)
270  //
271  // Compute the uncorrected pixel position (xSrc, ySrc):
272  //
273  // xSrc = xCenter + (dxRad + dxTan) * maxDist
274  // ySrc = yCenter + (dyRad + dyTan) * maxDist
275  //
276  // Mathematical definitions and restrictions:
277  //
278  // Let { xSrc, ySrc } = f (xDst, yDst) be the warp function defined
279  // above.
280  //
281  // Let xSrc = fx (xDst, yDst) be the x-component of the warp function.
282  // Let ySrc = fy (xDst, yDst) be the y-component of the warp function.
283  //
284  // f (x, y) must be an invertible function.
285  //
286  // fx (x, y) must be an increasing function of x.
287  // fy (x, y) must be an increasing function of x.
288  //
289  // The parameters kr0, kr1, kr2, and kr3 must define an increasing
290  // radial warp function. Specifically, let w (r) be the radial warp
291  // function:
292  //
293  // w (r) = (kr0 * r) + (kr1 * r^3) + (kr2 * r^5) + (kr3 * r^7).
294  //
295  // w (r) must be an increasing function.
296 
297  dng_vector fRadParams [kMaxColorPlanes];
298  dng_vector fTanParams [kMaxColorPlanes];
299 
300  public:
301

356 class dng_warp_params_fisheye: public dng_warp_params
357  {
358 
359  public:
360 
361  // Radial warp coefficients. These define a warp from corrected pixel
362  // coordinates (xDst, yDst) to uncorrected pixel coordinates (xSrc,
363  // ySrc) for each plane P as follows:
364  //
365  // Let kr0 = fRadParams [P][0]
366  // kr1 = fRadParams [P][1]
367  // kr2 = fRadParams [P][2]
368  // kr3 = fRadParams [P][3]
369  //
370  // Let (xCenter, yCenter) be the optical image center (see fCenter,
371  // below) expressed in pixel coordinates. Let maxDist be the Euclidean
372  // distance (in pixels) from (xCenter, yCenter) to the farthest image
373  // corner.
374  //
375  // First, compute the normalized distance of the corrected pixel
376  // position (xDst, yDst) from the image center:
377  //
378  // dx = (xDst - xCenter) / maxDist
379  // dy = (yDst - yCenter) / maxDist
380  //
381  // r = sqrt (dx^2 + dy^2)
382  //
383  // Compute the radial correction term:
384  //
385  // t = atan (r)
386  //
387  // rWarp = (kr0 * t) + (kr1 * t^3) + (kr2 * t^5) + (kr3 * t^7)
388  //
389  // ratio = rWarp / r
390  //
391  // dxRad = dx * ratio
392  // dyRad = dy * ratio
393  //
394  // Compute the uncorrected pixel position (xSrc, ySrc):
395  //
396  // xSrc = xCenter + (dxRad * maxDist)
397  // ySrc = yCenter + (dyRad * maxDist)
398  //
399  // The parameters kr0, kr1, kr2, and kr3 must define an increasing
400  // radial warp function. Specifically, let w (r) be the radial warp
401  // function:
402  //
403  // t = atan (r)
404  //
405  // w (r) = (kr0 * t) + (kr1 * t^3) + (kr2 * t^5) + (kr3 * t^7).
406  //
407  // w (r) must be an increasing function.
